what is Structure in c ?
--> Structure is a user defined data type
--> Structure is a collection of different data types


int  char  float  double

syntax:

    struct structure_name
    {
        data_type member1;
        data_type member2;
        data_type member3;
    };

    struct student
    {
        int roll_no;
        char name[20];
        float marks;
    };
